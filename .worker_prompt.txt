### `admin/src/hooks/useManifest.ts`
```ts
export function useManifest(platform: 'mobile' = 'mobile') {
  const { data, isLoading } = useQuery({
    queryKey: ['manifests', platform],
    queryFn: () => api.get(`/admin/manifests?platform=${platform}`),
  });
  // select active version + draft version
  return { active, draft, versions: data, isLoading };
}
```
