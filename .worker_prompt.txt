### 3. Cashfree integration вҖ” `mobile/lib/payments.ts`
```typescript
import { CFPaymentGatewayService } from 'react-native-cashfree-pg-sdk';
import { CFSession, CFEnvironment, CFDropCheckoutPayment, CFThemeBuilder, CFPaymentComponentBuilder, CFPaymentComponent } from 'cashfree-pg-api-contract';
import { api } from './api';

export async function startTokenTopup(bundleId: string) {
  // 1. Create order on backend
  const { data: order } = await api.post('/billing/checkout/topup', { bundle_id: bundleId });
  // order shape: { order_id, order_token, amount, currency }

  // 2. Build session
  const session = new CFSession(
    order.order_token,
    order.order_id,
    CFEnvironment.SANDBOX, // switch to PRODUCTION for live
  );

  const theme = new CFThemeBuilder()
    .setBackgroundColor('#08090B')
    .setPrimaryTextColor('#F7F8F8')
    .setNavigationBarBackgroundColor('#111113')
    .setNavigationBarTextColor('#F7F8F8')
    .setButtonBackgroundColor('#7170FF')
    .setButtonTextColor('#FFFFFF')
    .build();

  const paymentComponents = new CFPaymentComponentBuilder()
    .add(CFPaymentComponent.CFPaymentModes.UPI)
    .add(CFPaymentComponent.CFPaymentModes.CARD)
    .add(CFPaymentComponent.CFPaymentModes.NB)
    .add(CFPaymentComponent.CFPaymentModes.WALLET)
    .build();

  const dropPayment = new CFDropCheckoutPayment(session, paymentComponents, theme);

  CFPaymentGatewayService.setCallback({
    onVerify(orderId) { console.log('Payment verified', orderId); },
    onError(error, orderId) { console.log('Payment error', error, orderId); },
  });

  CFPaymentGatewayService.doPayment(dropPayment);
}

export async function startSubscription(planId: string) {
  const { data } = await api.post('/billing/checkout/subscribe', { plan_id: planId });
  // Open authorization_link in webview or browser
  return data;
}
```
